1. 不借助临时变量，进行两个整数的交换

```
  function swap(a , b) {  
    b = b - a; // 借助两者的差值
    a = a + b; // 此时相当于 a = a + b - a 实现了原来的a值被赋予原来的b值
    b = a - b; // 此时 b = a + b - a - b + a 实现了原来的b值被赋予原来的a值
    return [a,b];
  }
  let a = 3;
  let b = 4;
  let res = swap(a, b);
  console.log('---------question 1--------');  
  console.log(res); // [4,3]
  
```

4. 验证一个数是否是质数

```
var isPrime = function(n) {
    var divider = 2;
    var limit = Math.sqrt(n);
    while(divider<=limit) {
        if(n%divider == 0) {
            return false;
        }
        divider++;
    }
    return true;
}
```
